// 1. 健康档案管理
POST /api/health/profiles           // 创建/更新健康档案
GET /api/health/profiles             // 获取用户健康档案列表
GET /api/health/profiles/current     // 获取当前健康档案
DELETE /api/health/profiles/:id      // 删除健康档案

// 2. 身体指标记录
POST /api/health/metrics             // 记录身体指标
GET /api/health/metrics             // 获取身体指标历史
GET /api/health/metrics/:date        // 获取特定日期指标

// 3. 营养目标计算
POST /api/health/nutrition-goals     // 计算并保存营养目标
GET /api/health/nutrition-goals/current // 获取当前营养目标

// 4. 健康目标管理
POST /api/health/objectives          // 设置健康目标
GET /api/health/objectives           // 获取健康目标列表
PUT /api/health/objectives/:id       // 更新目标进度

// 5. 健康评估
POST /api/health/assessments         // 提交健康评估
GET /api/health/assessments          // 获取评估历史
